% 1. First Order ODE
% --------------------------------------------
\begin{studybox}{一阶微分方程 / First Order ODE}
    \textbf{[CN] 定义}: 
    涉及函数 $y$ 及其一阶导数 $y'$ 的方程。
    \begin{itemize}[leftmargin=*]
        \item \textbf{分离变量法}: 将 $x$ 和 $y$ 分别移到等号两边求解。
        \item \textbf{线性方程}: $y' + p(x)y = q(x)$。
    \end{itemize}
    \tcblower
    \textbf{[EN] Definition}: 
    Equation involving independent variable $x$, function $y$, and derivative $y'$.
    \begin{itemize}[leftmargin=*]
        \item \textbf{Separable}: Can be rewritten as $f(y)dy = g(x)dx$.
        \item \textbf{Linear}: Standard form $y' + p(x)y = q(x)$. Integrating factor method.
    \end{itemize}
\end{studybox}

\begin{formulabox}
\textbf{二阶线性常系数 ODE}:
\begin{equation}
    ay'' + by' + cy = 0
\end{equation}
特征方程: $ar^2 + br + c = 0$.
解的形式取决于判别式 $\Delta = b^2 - 4ac$:
\begin{itemize}
    \item $\Delta > 0$: $y = C_1 e^{r_1 x} + C_2 e^{r_2 x}$ (Overdamped)
    \item $\Delta = 0$: $y = C_1 e^{rx} + C_2 x e^{rx}$ (Critically Damped)
    \item $\Delta < 0$: $y = e^{\alpha x}(C_1 \cos \beta x + C_2 \sin \beta x)$ (Underdamped)
\end{itemize}
\end{formulabox}

\begin{thesisbox}[论文关联 / Project Application]
    \textbf{[CN]}: 
    物理系统通常由微分方程建模：
    \begin{itemize}
        \item \textbf{RC 电路}: 电容充电过程遵循一阶 ODE ($RC \frac{dV}{dt} + V = V_{in}$)。解是指数衰减函数。
        \item \textbf{机械振动}: 加速度 ($a=x''$) 与力 ($F=-kx-bv$) 的关系构成二阶 ODE ($mx'' + bx' + kx = 0$)。这解释了\textbf{过阻尼}和\textbf{欠阻尼}现象。
    \end{itemize}

    \tcblower

    \textbf{[EN]}: 
    \begin{itemize}
        \item \textbf{RC Circuit}: Modeled by 1st order ODE. Solution involves time constant $\tau=RC$.
        \item \textbf{Vibration}: Mass-Spring-Damper system is a 2nd order ODE ($mx'' + bx' + kx = 0$). Explains Overdamped vs Underdamped responses in sensors.
    \end{itemize}
\end{thesisbox}
