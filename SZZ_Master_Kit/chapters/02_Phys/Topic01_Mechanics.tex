\section{Kinematics \& Dynamics (PH1)}

\begin{tcolorbox}[colback=yellow!10!white,colframe=yellow!50!black,title=\textbf{Exam Cheat Sheet / 极速复习}]
\begin{itemize}
    \item \textbf{Newton's 2nd Law}: $\vec{F} = \frac{d\vec{p}}{dt} = m\vec{a}$.
    \item \textbf{Conservation}: Verify if External Forces = 0 (Momentum conserved) or Non-conservative work = 0 (Energy conserved).
    \item \textbf{Rigid Body}: Rotation analogue ($F \to \tau$, $m \to I$, $v \to \omega$).
\end{itemize}
\end{tcolorbox}

\subsection{Concepts / 核心概念}
\begin{keywords}
\textbf{Force, Momentum, Work, Kinetic/Potential Energy, Moment of Inertia}
\end{keywords}

\begin{concept}{Harmonic Oscillator / 谐振子}
The mechanical analogue to the LC circuit.
\begin{itemize}
    \item **Spring Force**: $F = -kx$.
    \item **Equation**: $m \ddot{x} + kx = 0 \implies \ddot{x} + \omega_0^2 x = 0$.
    \item **Solution**: $x(t) = A \cos(\omega_0 t + \phi)$.
\end{itemize}
\end{concept}

\subsection{Formulas / 公式}
\begin{equation}
    E_k = \frac{1}{2}mv^2, \quad E_p = mgh \quad \text{(Point Mass)}
\end{equation}
\begin{equation}
    \vec{\tau} = \vec{r} \times \vec{F} \quad \text{(Torque)}
\end{equation}

\subsection{Exam Questions / 常考题型}
\begin{itemize}
    \item "A ball rolls down a slope. Calculate velocity at bottom." (Use Energy Conservation: $mgh = \frac{1}{2}mv^2 + \frac{1}{2}I\omega^2$).
    \item "Define Moment of Inertia." (A: Resistance to change in rotational motion. $\int r^2 dm$).
\end{itemize}
