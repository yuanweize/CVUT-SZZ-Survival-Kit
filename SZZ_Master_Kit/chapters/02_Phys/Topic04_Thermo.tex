% 1. Laws of Thermodynamics
% --------------------------------------------
\begin{studybox}{热力学定律 / Laws of Thermodynamics}
    \textbf{[CN] 定义}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{第零定律}: 如果 A 和 B 平衡，B 和 C 平衡，则 A 和 C 平衡 (温度定义)。
        \item \textbf{第一定律}: 能量守恒。内能的变化等于吸收的热量加上外界对系统做的功 ($\Delta U = Q + W$)。
        \item \textbf{第二定律}: 孤立系统的熵 (Entropy) 永不减少。热量不能自发从低温流向高温。
    \end{itemize}
    \tcblower
    \textbf{[EN] Definition}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{0th Law}: Defines Temperature (Thermal Equilibrium).
        \item \textbf{1st Law}: Conservation of Energy. $\Delta U = Q + W$.
        \item \textbf{2nd Law}: Entropy of an isolated system never decreases. Heat cannot spontaneously flow from cold to hot.
    \end{itemize}
\end{studybox}

\begin{formulabox}
\textbf{理想气体状态方程 / Ideal Gas Law}:
\begin{equation}
    pV = nRT = NkT
\end{equation}
$p$: 压强 (pressure), $V$: 体积 (volume), $T$: 绝对温度 (absolute temperature, Kelvin).

\textbf{热传递 / Heat Transfer}:
\begin{itemize}
    \item \textbf{[CN]} 传导 (Conduction): 接触传递 (Fourier's Law) / \textbf{[EN]} Contact transfer
    \item \textbf{[CN]} 对流 (Convection): 流体流动传递 / \textbf{[EN]} Fluid motion transfer
    \item \textbf{[CN]} 辐射 (Radiation): 电磁波传递 (Stefan-Boltzmann Law) / \textbf{[EN]} EM wave transfer
\end{itemize}
\end{formulabox}

\begin{warnbox}[考试陷阱 / Exam Pitfalls]
    \begin{itemize}
        \item \textbf{[CN]} Kelvin Scale: 热力学公式永远使用开尔文 ($T_K = T_C + 273.15$)\\
              \textbf{[EN]} Kelvin Scale: Thermodynamics formulas ALWAYS use Kelvin!
        \item \textbf{[CN]} Adiabatic vs Isothermal: 绝热过程 ($Q=0$) 不等于等温过程 ($T=\text{const}$)\\
              \textbf{[EN]} Adiabatic ($Q=0$) $\neq$ Isothermal ($T=\text{const}$)
        \item \textbf{[CN]} Entropy meaning: 熵是系统无序度 (Disorder) 的度量\\
              \textbf{[EN]} Entropy: measure of system's disorder
    \end{itemize}
\end{warnbox}
