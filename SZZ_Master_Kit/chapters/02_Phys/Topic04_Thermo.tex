% 1. Laws of Thermodynamics
% --------------------------------------------
\begin{studybox}{热力学定律 / Laws of Thermodynamics}
    \textbf{[CN] 定义}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{第零定律}: 如果 A 和 B 平衡，B 和 C 平衡，则 A 和 C 平衡 (温度定义)。
        \item \textbf{第一定律}: 能量守恒。内能的变化等于吸收的热量加上外界对系统做的功 ($\Delta U = Q + W$)。
        \item \textbf{第二定律}: 孤立系统的熵 (Entropy) 永不减少。热量不能自发从低温流向高温。
    \end{itemize}
    \tcblower
    \textbf{[EN] Definition}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{0th Law}: Defines Temperature (Thermal Equilibrium).
        \item \textbf{1st Law}: Conservation of Energy. $\Delta U = Q + W$.
        \item \textbf{2nd Law}: Entropy of an isolated system never decreases. Heat cannot spontaneously flow from cold to hot.
    \end{itemize}
\end{studybox}

\begin{formulabox}
\textbf{理想气体状态方程 / Ideal Gas Law}:
\begin{equation}
    pV = nRT = NkT
\end{equation}
$p$: 压强, $V$: 体积, $T$: 绝对温度 (Kelvin).

\textbf{热传递 / Heat Transfer}:
\begin{itemize}
    \item \textbf{传导 (Conduction)}: 接触传递 (Fourier's Law).
    \item \textbf{对流 (Convection)}: 流体流动传递.
    \item \textbf{辐射 (Radiation)}: 电磁波传递 (Stefan-Boltzmann Law).
\end{itemize}
\end{formulabox}

\begin{thesisbox}[论文关联 / Project Application]
    \textbf{[CN]}: 
    \begin{itemize}
        \item \textbf{ESP32 发热}: 芯片工作时消耗电能转换为热能 (焦耳热 $P=IV$)。
        \item \textbf{散热设计}: PCB 的铜箔层充当散热片 (Heatsink)，通过\textbf{热传导}将热量散发。若封装密闭，主要依靠\textbf{热辐射}散发。
        \item \textbf{BMP180 传感器}: 测量环境温度，依据其实是电阻随温度的变化 (热敏电阻原理) 或带隙电压基准特性。
    \end{itemize}

    \tcblower

    \textbf{[EN]}: 
    \begin{itemize}
        \item \textbf{Self-Heating}: ESP32 converts electrical energy into heat ($P_{loss} \approx IV$).
        \item \textbf{Heat Dissipation}: The PCB copper layers act as heatsinks, dissipating heat via \textbf{Conduction}.
        \item \textbf{Temperature Sensing}: BMP180 measures temperature, critical for compensating MEMS bias drift (which is temperature-dependent).
    \end{itemize}
\end{thesisbox}

\begin{warnbox}[考试陷阱 / Exam Pitfalls]
    \begin{itemize}
        \item \textbf{Kelvin Scale}: 热力学公式永远使用开尔文 ($T_K = T_C + 273.15$)。
        \item \textbf{Adiabatic vs Isothermal}: 绝热过程 ($Q=0$) 不等于等温过程 ($T=\text{const}$)。
        \item \textbf{Entropy meaning}: 熵是系统无序度 (Disorder) 的度量。
    \end{itemize}
\end{warnbox}
