% --------------------------------------------
% 1. Wave Properties
% --------------------------------------------
\begin{studybox}{波的特性 / Wave Properties}
    \textbf{[CN] 定义}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{干涉 (Interference)}: 波的叠加。同相增强 (Constructive)，反相抵消 (Destructive)。
        \item \textbf{衍射 (Diffraction)}: 波绕过障碍物的现象。孔越小，衍射越明显。
        \item \textbf{偏振 (Polarization)}: 横波振动方向的限制 (仅适用于横波，如光)。
    \end{itemize}
    \tcblower
    \textbf{[EN] Definition}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{Interference}: Superposition of waves. Constructive (in-phase) or Destructive (out-of-phase).
        \item \textbf{Diffraction}: Bending of waves around obstacles. Significant when $\lambda \approx$ aperture size.
        \item \textbf{Polarization}: Orientation of oscillation (Transverse waves only).
    \end{itemize}
\end{studybox}

\begin{formulabox}
\textbf{电磁波谱 / EM Spectrum}:
Radio $\to$ Microwave $\to$ IR $\to$ Visible $\to$ UV $\to$ X-Ray $\to$ Gamma.

\textbf{光速的关系}:
\begin{equation}
    c = \lambda f \approx 3 \times 10^8 \, \text{m/s}
\end{equation}
\end{formulabox}

% Thesis Connection

\begin{thesisbox}[论文关联 / Project Application]
    \textbf{[CN]}: 
    \begin{itemize}
        \item \textbf{Wi-Fi 通信}: ESP32 使用 2.4 GHz 载波。
        波长计算: $\lambda = \frac{c}{f} = \frac{3 \cdot 10^8}{2.4 \cdot 10^9} \approx 0.125 \, \text{m} = 12.5 \, \text{cm}$。
        \item \textbf{天线设计}: 最优天线长度通常是 $\lambda/4$ (约 3.1 cm)。这也是为什么 ESP32 模块上的 PCB 天线是这个尺寸。
        \item \textbf{多径干扰}: 室内环境中，Wi-Fi 信号反射形成多径效应 (Multipath)，产生干扰。
    \end{itemize}

    \tcblower

    \textbf{[EN]}: 
    \begin{itemize}
        \item \textbf{Wi-Fi Carrier}: 2.4 GHz ISM band.
        \item \textbf{Wavelength}: $\lambda \approx 12.5$ cm. Optimum antenna length is typically $\lambda/4$ ($\approx 3.1$ cm), explaining the size of the PCB trace antenna.
        \item \textbf{Multipath}: Reflection causes signal interference (Fading) in indoor environments.
    \end{itemize}
\end{thesisbox}

\begin{warnbox}[考试陷阱 / Exam Pitfalls]
    \begin{itemize}
        \item \textbf{Medium Requirement}: 电磁波不需要介质 (可以在真空中传播)。声波需要介质。
        \item \textbf{Index of Refraction}: $n = c/v$. 光在介质中速度减慢。
    \end{itemize}
\end{warnbox}
