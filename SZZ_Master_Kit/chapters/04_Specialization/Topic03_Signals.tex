\section{Signals \& Systems (TES)}

\begin{tcolorbox}[colback=yellow!10!white,colframe=yellow!50!black,title=\textbf{Exam Cheat Sheet / 极速复习}]
\begin{itemize}
    \item \textbf{Transforms}: Laplace ($s$-domain) for Continuous; Z-transform ($z$-domain) for Discrete.
    \item \textbf{Sampling}: $f_s \ge 2 f_{max}$ (Nyquist). Aliasing happens if violated.
    \item \textbf{Convolution}: Output $y(t) = x(t) * h(t)$.
    \item \textbf{Thesis Link}: Your accelerometer data is sampled at discrete intervals (e.g., 50Hz).
\end{itemize}
\end{tcolorbox}

\subsection{Concepts / 核心概念}
\begin{keywords}
\textbf{LTI Systems, Impulse Response, Frequency Response, Pole-Zero Plot, DFT/FFT}
\end{keywords}

\begin{concept}{LTI Systems / 线性时不变系统}
Linear Time-Invariant systems are characterized completely by their **Impulse Response** $h(t)$.
\begin{itemize}
    \item \textbf{Linear}: $S(a x_1 + b x_2) = a S(x_1) + b S(x_2)$.
    \item \textbf{Time-Invariant}: Shift in input causes same shift in output.
\end{itemize}
\end{concept}

\begin{concept}{Fourier Series \& Transform / 傅里叶变换}
Decomposing a signal into sine waves.
\begin{itemize}
    \item \textbf{Periodic}: Fourier Series (discrete spectrum).
    \item \textbf{Aperiodic}: Fourier Transform (continuous spectrum).
\end{itemize}
\end{concept}

\subsection{Formulas / 公式}
\begin{equation}
    X(\omega) = \int_{-\infty}^{\infty} x(t) e^{-j\omega t} dt \quad \text{(CTFT)}
\end{equation}
\begin{equation}
    y[n] = \sum_{k=-\infty}^{\infty} x[k] h[n-k] \quad \text{(Discrete Convolution)}
\end{equation}

\subsection{Exam Questions / 常考题型}
\begin{itemize}
    \item "What happens if you sample a 100Hz signal at 150Hz?" (A: Aliasing. You "see" a 50Hz ghost signal. $150 - 100 = 50$).
    \item "Explain the relationship between the s-plane and z-plane." (A: Left Half Plane in s $\leftrightarrow$ Unit Circle Interior in z).
\end{itemize}
