% 1. Transformers
% --------------------------------------------
\begin{studybox}{变压器 / Transformer}
    \textbf{[CN] 定义}: 
    基于电磁感应，只转换交流电压/电流，不改变频率。
    $V_1/V_2 = N_1/N_2$。
    \tcblower
    \textbf{[EN] Definition}: 
    Static device transferring AC energy via induction. 
    Steps up/down voltage ($V_p/V_s = N_p/N_s$) without changing frequency.
\end{studybox}

% --------------------------------------------
% 2. Motors
% --------------------------------------------
\begin{studybox}{电动机 / Electric Motors}
    \textbf{[CN] 定义}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{直流电机 (DC Motor)}: 使用换向器 (Brushed) 或电子换向 (BLDC)。速度 $\propto$ 电压。
        \item \textbf{感应电机 (Induction Motor)}: 异步。转子速度总是落后于旋转磁场 (Slip)。
        \item \textbf{同步电机 (Synchronous Motor)}: 转子速度锁定于电源频率。
    \end{itemize}
    \tcblower
    \textbf{[EN] Definition}: 
    \begin{itemize}[leftmargin=*]
        \item \textbf{DC Motor}: Speed $\propto$ Voltage. Brushed or Brushless (BLDC).
        \item \textbf{Induction (Asynchronous)}: Rotor "slips" behind the rotating magnetic field.
        \item \textbf{Synchronous}: Rotor rotates in lock-step with supply frequency.
    \end{itemize}
\end{studybox}

\begin{thesisbox}[论文关联 / Project Application]
    \textbf{[CN]}: 
    虽然你的项目主要是传感器节点，但智能家居通常控制电机负载：
    \begin{itemize}
        \item \textbf{继电器控制}: 你的 ESP32 可以驱动继电器来开启/关闭大功率交流电机（如风扇、水泵）。
        \item \textbf{感性负载}: 电机是感性负载，关闭时会产生反电动势 (Back EMF)。需要反向二极管 (Flyback Diode) 保护电路。
    \end{itemize}

    \tcblower

    \textbf{[EN]}: 
    Smart homes often control motors:
    \begin{itemize}
        \item \textbf{Relays}: ESP32 drives relays to switch AC motors (fans, pumps).
        \item \textbf{Inductive Load}: Motors generate Back EMF when switched off. A Flyback Diode is required to protect the control circuitry.
    \end{itemize}
\end{thesisbox}
